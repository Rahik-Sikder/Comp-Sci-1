public class SeventhGradeMathMethods {
     public static void main (String args[]){
     
     // Math.sqrt(a)
     // Returns the square root of a
     // input: double
     // output: double
     
     System.out.println("The sqrt of 70 is: " + Math.sqrt(70));
         
        
     
     // Math.abs(a)
     // Returns the absolute value of a
     // input: double, float, int, long
     // output: double, float, int, long
     // OVERLOADED
     
     System.out.println("The abs of -10 is: " + Math.abs(-10));
     
     
     
     // Math.pow(base, exponent)
     // Returns the base raised to the exponent
     // input: double, double
     // output: double
     
     System.out.println("5^3 is: " + Math.pow(5, 3));
     
     
     
     // Data values in the math class
     // classes can store methods (like above) and also variables
     
     // Math.PI - 3.14...
     // ratio between circumference of a cirlce and the diameter of a cirle
     System.out.println(Math.PI);
     
     // Math.E - 2.71...
     // base of the natural log
     System.out.println(Math.E);
     
     
     
     }
}
