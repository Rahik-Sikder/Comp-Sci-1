public class EscapeSequences {
    public static void main(String args[]){
    
        // Michael said, "Theres always money in the banana stand."
        
        
        System.out.println("Michael said, \"Theres always money in the banana stand.\"");
        
        // double quote excape - \"
        // allows us to print a " instead of ending a string.
        
        
        // backslash escape - \\
        // allows us to print a \ instead of starting an escape.
        System.out.println("The file is located at H:\\CompSci");
        
        
        // new line escape - \n
        // prints an end of line character
        System.out.println("this is on \ntwo lines.");
        
        
        
        
        
    
    }
}
