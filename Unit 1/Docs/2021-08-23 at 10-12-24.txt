public class Shortcuts{
    public static void main (String args[]){
    
    // Shortcuts are used to change the value of a variable
    
    int num = 10;
    
    // increase by 15    
    num += 15;
    System.out.println(num);
    
        
    //Decrease bt 5
    num -= 5;
    System.out.println(num);
    
    
    // Increase by factor of 3
    num *= 3;
    System.out.println(num);
    
    
    // Decrease by factor of 5
    num /= 5;
    System.out.println(num);
    
    
    // Find the remainder after deviding by 2
    num %= 2;
    
    
    // Unary - to increase by 1
    num++;
    
    
    }
}
