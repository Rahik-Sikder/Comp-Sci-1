public class StringLoops {
    public static void main (String args[]){
    
        // Printing a string one letter at a time using a loop
        
        String word = "computer";
        //             01234567
        
        for (int i = 0; i < word.length(); i++){
            System.out.print(word.substring(i, i+1);
        }
        
    
    }
}
