public class VariablesExample {
    public static void main(String[] args){
        
        // 1. Declaring the variable
        
        int x;
        // int is the type of variable
        // num is the variable name
        
        // 2. initializing the variable (storing a value)
        
        x = 10;
        
        System.out.println(x);
        
        int y = 9;
        
        System.out.println(y);
        
        // Using variables for math
        
        System.out.println(x+y);
        
        // Reintialize a variable
        
        y = 100;
        
        // Declare and initialize on the same line 
        
        int z = 7;
        
        // There are 30 students in the class.
        // ^ want to print this but make the number 30 come from a variabel
        
        int students = 30;
        
        System.out.println("There are " + students + " in the class");
        
    }
}
